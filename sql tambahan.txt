  AUTO_INCREMENT, `id_guru` INT(11) NOT NULL, `status`

  NULL, `keterangan` TEXT NULL, CONSTRAINT `pk_absensi_guru` PRIMARY
  KEY(`id_absensi`) ) DEFAULT CHARACTER SET = utf8 COLLATE = utf8_general_ci
